    fn find_error_nums(mut nums: Vec<i32>) -> Vec<i32> {
            let mut duplicate = 0;
            let mut missing = 0;

            // find duplicate
            for i in 0..nums.len() {
                let index = nums[i].abs() as usize - 1;

                if nums[index] < 0 {
                    duplicate = nums[i].abs();
                } else {
                    nums[index] = -nums[index];
                }
            }

            // find missing
            for i in 0..nums.len() {
                if nums[i] > 0 {
                    missing = (i + 1) as i32;
                    break;
                }
            }

            vec![duplicate, missing]
    }
